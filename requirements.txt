# --- Broker data (Angel One SmartAPI, read-only) ---
# We use Angel One purely for live/EOD market data (no order placement),
# and we call the REST API directly with `requests` + `pyotp` instead of
# pulling in `smartapi-python`. Direct REST keeps the dependency surface
# small and is much easier to mock in tests; pyotp generates the rolling
# 6-digit TOTP from the base32 secret you set up at app registration.
pyotp==2.9.0
# websocket-client powers the live LTP stream (Angel One SmartStream WS 2.0).
# We talk the WS protocol directly (one small, well-maintained lib) rather than
# pulling the whole smartapi-python SDK, which pins conflicting versions of
# requests/pyotp/websocket-client. The dashboard's live prices subscribe to
# tick-by-tick LTP here; REST getLtpData remains only as a cold-start fallback.
websocket-client==1.8.0

# --- Web dashboard (Phase A: local; later containerised on Oracle Cloud) ---
# FastAPI + Starlette give us async views, dependency injection, and
# typed request/response models that line up with our Pydantic contracts.
# Uvicorn[standard] pulls in httptools + uvloop on POSIX (silent no-op on
# Windows) for measurable throughput improvement; on Windows it falls
# back to the asyncio default reactor without complaint.
# itsdangerous signs the session cookie. We never use it for free-form
# storage -- only the user's authenticated state -- but we still want
# tamper-evident signing so a stolen cookie expires when we rotate
# WEB_SESSION_SECRET in .env.
# python-multipart is required by FastAPI's Form() parsing for the login
# POST. starlette pulls it in optionally; pinning here makes the install
# deterministic on fresh venvs.
fastapi==0.115.5
uvicorn[standard]==0.32.1
itsdangerous==2.2.0
python-multipart==0.0.17
httpx==0.27.2          # used by tests/test_web.py via TestClient

# --- Cloud mirror DB (Neon Postgres) ---
# psycopg 3 with the binary extra ships prebuilt wheels for Windows (local
# publish step) and Linux (Render dashboard reader), so no system libpq is
# needed. Only used when DATABASE_URL is set; the local pipeline stays on
# SQLite. The connection string is read from the DATABASE_URL env var --
# never hardcoded -- and Neon requires sslmode=require (TLS) by default.
psycopg[binary]==3.2.3
